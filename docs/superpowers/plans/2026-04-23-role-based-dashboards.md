# Role-Based Dashboards Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build three role-scoped dashboard endpoints (Owner, Technician, Front-Desk) that return fully typed DTOs derived from existing tables, backed by a composable service layer with WebSocket-driven invalidation.

**Architecture:** Thin route handlers compose pure metric functions from `dashboard.service.ts`. A `Scope` object (built by middleware from the session + `ShopSettings.timezone`) is the single authorization boundary. Job mutations call `emitDashboardChanged()` to fan out `dashboard:invalidate` WS events to the relevant role/technician predicates.

**Tech Stack:** Fastify, Prisma 7, PostgreSQL 17, Better Auth (for session + permissions), vitest, existing `wsBroadcast` plugin.

**Spec:** `docs/superpowers/specs/2026-04-23-role-based-dashboards-design.md`

import type { Decimal } from "@prisma/client/runtime/library";

export type DateRange = { start: Date; end: Date };

function zonedParts(tz: string, d: Date) {
  const fmt = new Intl.DateTimeFormat("en-US", {
    timeZone: tz,
    year: "numeric",
    month: "2-digit",
    day: "2-digit",
    hour: "2-digit",
    minute: "2-digit",
    second: "2-digit",
    hour12: false,
  });
  const parts = Object.fromEntries(
    fmt.formatToParts(d).map((p) => [p.type, p.value])
  );
  return {
    year: Number(parts.year),
    month: Number(parts.month),
    day: Number(parts.day),
  };
}

function utcFromZonedYmd(tz: string, y: number, m: number, d: number): Date {
  // Compute the UTC instant for {y, m, d} 00:00:00 in `tz` by bisecting from UTC.
  const utcGuess = new Date(Date.UTC(y, m - 1, d, 0, 0, 0));
  const parts = zonedParts(tz, utcGuess);
  const diff =
    (parts.year - y) * 365 * 24 * 60 +
    (parts.month - m) * 31 * 24 * 60 +
    (parts.day - d) * 24 * 60;
  // Approximate correction, then refine once more.
  const corrected = new Date(utcGuess.getTime() - diff * 60_000);
  const parts2 = zonedParts(tz, corrected);
  if (parts2.year !== y || parts2.month !== m || parts2.day !== d) {
    // Fall back to iterative search in a narrow window (±26h covers any DST).
    for (let offMin = -26 * 60; offMin <= 26 * 60; offMin += 15) {
      const t = new Date(utcGuess.getTime() + offMin * 60_000);
      const p = zonedParts(tz, t);
      if (p.year === y && p.month === m && p.day === d) return t;
    }
  }
  return corrected;
}

export function todayRange(tz: string, now: Date = new Date()): DateRange {
  const { year, month, day } = zonedParts(tz, now);
  const start = utcFromZonedYmd(tz, year, month, day);
  const end = new Date(start.getTime() + 24 * 60 * 60 * 1000);
  return { start, end };
}

export function monthRange(tz: string, now: Date = new Date()): DateRange {
  const { year, month } = zonedParts(tz, now);
  const start = utcFromZonedYmd(tz, year, month, 1);
  const nextY = month === 12 ? year + 1 : year;
  const nextM = month === 12 ? 1 : month + 1;
  const end = utcFromZonedYmd(tz, nextY, nextM, 1);
  return { start, end };
}

export function toMoney(d: Decimal | number | null | undefined): number {
  if (d === null || d === undefined) return 0;
  const n = typeof d === "number" ? d : Number(d.toString());
  return Math.round(n * 100) / 100;
}
```

import type { DashboardRole, Scope } from "@shared/types/dashboard";
import type { FastifyReply, FastifyRequest } from "fastify";

const TTL_MS = 60_000;
let cached: { value: string; expiresAt: number } | null = null;

export function __resetTzCache() {
  cached = null;
}

async function getShopTimezone(req: FastifyRequest): Promise<string> {
  const now = Date.now();
  if (cached && cached.expiresAt > now) return cached.value;
  const row = await req.server.prisma.shopSettings.findFirst({
    select: { timezone: true },
  });
  const value = row?.timezone ?? process.env.TZ ?? "UTC";
  cached = { value, expiresAt: now + TTL_MS };
  return value;
}

declare module "fastify" {
  interface FastifyRequest {
    dashboardScope?: Scope;
  }
}

export async function dashboardScope(
  req: FastifyRequest,
  reply: FastifyReply
) {
  if (!req.user) {
    await reply.status(401).send({ error: "Authentication required" });
    return;
  }
  const shopTz = await getShopTimezone(req);
  req.dashboardScope = {
    role: req.user.role as DashboardRole,
    userId: req.user.id,
    shopTz,
  };
}
```

import type { FastifyReply, FastifyRequest } from "fastify";

export function requireRoles(...allowed: string[]) {
  const set = new Set(allowed);
  return async (req: FastifyRequest, reply: FastifyReply) => {
    if (!req.user) {
      await reply.status(401).send({ error: "Authentication required" });
      return;
    }
    if (!set.has(req.user.role)) {
      await reply.status(403).send({ error: "Insufficient permissions" });
      return;
    }
  };
}

import type { JobStatus, PrismaClient } from "@prisma/client";
import type { Scope } from "@shared/types/dashboard";
import type { DateRange } from "../utils/time-range.js";

const ALL_STATUSES: JobStatus[] = [
  "INTAKE", "WAITING_FOR_PARTS", "IN_REPAIR", "ON_HOLD",
  "DONE", "DELIVERED", "RETURNED", "CANCELLED",
];

function scopeWhere(scope: Scope) {
  return scope.role === "TECHNICIAN" ? { technicianId: scope.userId } : {};
}

export async function pipelineCounts(
  prisma: PrismaClient,
  scope: Scope
): Promise<Record<JobStatus, number>> {
  const groups = await prisma.job.groupBy({
    by: ["status"],
    _count: { _all: true },
    where: scopeWhere(scope),
  });
  const counts = Object.fromEntries(
    ALL_STATUSES.map((s) => [s, 0])
  ) as Record<JobStatus, number>;
  for (const g of groups) counts[g.status] = g._count._all;
  return counts;
}

export async function activeJobsCount(
  prisma: PrismaClient,
  scope: Scope
): Promise<number> {
  return prisma.job.count({
    where: {
      ...scopeWhere(scope),
      status: { in: ["INTAKE", "IN_REPAIR", "ON_HOLD", "WAITING_FOR_PARTS"] },
    },

import { toMoney } from "../utils/time-range.js";

type RevCostRow = { revenue: string; cost: string };

async function revenueAndCost(
  prisma: PrismaClient,
  scope: Scope,
  range: DateRange
): Promise<{ revenue: number; cost: number }> {
  const techFilter =
    scope.role === "TECHNICIAN"
      ? `AND j."technicianId" = $3`
      : "";
  const params: unknown[] = [range.start, range.end];
  if (scope.role === "TECHNICIAN") params.push(scope.userId);

  const rows = await prisma.$queryRawUnsafe<RevCostRow[]>(
    `SELECT
       COALESCE((SELECT SUM("price") FROM "job_repairs" jr WHERE jr."jobId" = j."id"), 0)
       + COALESCE((SELECT SUM("totalCost") FROM "job_parts" jp WHERE jp."jobId" = j."id"), 0)
         AS revenue,
       COALESCE((SELECT SUM("totalCost") FROM "job_parts" jp WHERE jp."jobId" = j."id"), 0)
         AS cost
     FROM "jobs" j
     WHERE j."status" = 'DELIVERED'
       AND j."updatedAt" >= $1 AND j."updatedAt" < $2
       ${techFilter}`,
    ...params
  );
  let revenue = 0;
  let cost = 0;
  for (const r of rows) {
    revenue += Number(r.revenue);
    cost += Number(r.cost);
  }
  return { revenue: toMoney(revenue), cost: toMoney(cost) };
}

export async function revenueThisMonth(
  prisma: PrismaClient,
  scope: Scope,
  month: DateRange
): Promise<number> {
  const { revenue } = await revenueAndCost(prisma, scope, month);
  return revenue;
}

export async function avgProfitMargin(
  prisma: PrismaClient,
  scope: Scope,
  month: DateRange
): Promise<number> {
  const { revenue, cost } = await revenueAndCost(prisma, scope, month);
  if (revenue === 0) return 0;
  return Math.round(((revenue - cost) / revenue) * 100) / 100;
}

import type { FinancialTrendPoint } from "@shared/types/dashboard";

export async function financialTrend(
  prisma: PrismaClient,
  scope: Scope,
  days: number
): Promise<FinancialTrendPoint[]> {
  const techClause = scope.role === "TECHNICIAN" ? `AND j."technicianId" = $2` : "";
  const params: unknown[] = [scope.shopTz];
  if (scope.role === "TECHNICIAN") params.push(scope.userId);
  const rows = await prisma.$queryRawUnsafe<
    Array<{ day: Date; revenue: string; cost: string }>
  >(
    `WITH series AS (
       SELECT generate_series(
         (date_trunc('day', now() AT TIME ZONE $1) - INTERVAL '${days - 1} days')::date,
         (date_trunc('day', now() AT TIME ZONE $1))::date,
         '1 day'::interval
       )::date AS day
     )
     SELECT s.day,
       COALESCE(SUM(jr."price"), 0) + COALESCE(SUM(jp."totalCost"), 0) AS revenue,
       COALESCE(SUM(jp."totalCost"), 0) AS cost
     FROM series s
     LEFT JOIN "jobs" j
       ON j."status" = 'DELIVERED'
      AND (j."updatedAt" AT TIME ZONE $1)::date = s.day
      ${techClause}
     LEFT JOIN "job_repairs" jr ON jr."jobId" = j."id"
     LEFT JOIN "job_parts"   jp ON jp."jobId" = j."id"
     GROUP BY s.day
     ORDER BY s.day ASC`,
    ...params
  );
  return rows.map((r) => ({
    date: new Date(r.day).toISOString().slice(0, 10),
    revenue: toMoney(Number(r.revenue)),
    cost: toMoney(Number(r.cost)),
  }));
}

export async function avgRepairTimeHours(
  prisma: PrismaClient,
  userId: string,
  days: number
): Promise<number> {
  const since = new Date(Date.now() - days * 86_400_000);
  const rows = await prisma.job.findMany({
    where: {
      technicianId: userId,
      status: "DELIVERED",
      updatedAt: { gte: since },
    },
    select: { createdAt: true, updatedAt: true },
  });
  if (rows.length === 0) return 0;
  const hours =
    rows.reduce(
      (acc, r) => acc + (r.updatedAt.getTime() - r.createdAt.getTime()),
      0
    ) / rows.length / 3_600_000;
  return Math.round(hours * 10) / 10;
}

import type { FastifyInstance } from "fastify";
import type { DashboardInvalidateEvent } from "@shared/types/dashboard";

export type DashboardTarget = "OWNER" | "FRONT_DESK" | { technicianId: string };

type WsClientLike = { role: string; userId: string };

const PAYLOAD: DashboardInvalidateEvent = { type: "dashboard:invalidate" };

export function emitDashboardChanged(
  app: FastifyInstance,
  targets: DashboardTarget[]
): void {
  const broadcast = (app as { wsBroadcast?: (
    predicate: (c: WsClientLike) => boolean,
    payload: Record<string, unknown>,
  ) => void }).wsBroadcast;
  if (!broadcast) return;

  const roles = new Set<string>();
  const techIds = new Set<string>();
  for (const t of targets) {
    if (t === "OWNER") roles.add("OWNER");
    else if (t === "FRONT_DESK") roles.add("FRONT_DESK");
    else techIds.add(t.technicianId);
  }

  for (const role of roles) {
    broadcast((c) => c.role === role, PAYLOAD);
  }
  for (const id of techIds) {
    broadcast(
      (c) => c.role === "TECHNICIAN" && c.userId === id,
      PAYLOAD
    );
  }
}
